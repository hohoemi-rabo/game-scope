import { cache } from 'react'
import { createServerClient } from './server'
import 'server-only'

interface SearchParams {
  query?: string
  platforms?: string[]
  minScore?: number
  maxScore?: number
  limit?: number
}

/**
 * ã‚²ãƒ¼ãƒ æ¤œç´¢é–¢æ•°
 * ã‚¿ã‚¤ãƒˆãƒ«ã€ãƒ—ãƒ©ãƒƒãƒˆãƒ•ã‚©ãƒ¼ãƒ ã€ã‚¹ã‚³ã‚¢ã§ãƒ•ã‚£ãƒ«ã‚¿ãƒªãƒ³ã‚°
 */
export const searchGames = cache(async ({
  query = '',
  platforms = [],
  minScore = 0,
  maxScore = 100,
  limit = 50,
}: SearchParams) => {
  const supabase = createServerClient()

  let dbQuery = supabase
    .from('games')
    .select('*')
    .gte('metascore', minScore)
    .lte('metascore', maxScore)

  // ã‚¿ã‚¤ãƒˆãƒ«æ¤œç´¢ï¼ˆæ—¥æœ¬èª/è‹±èªã®ä¸¡æ–¹ã§æ¤œç´¢ï¼‰
  if (query) {
    dbQuery = dbQuery.or(`title_ja.ilike.%${query}%,title_en.ilike.%${query}%`)
  }

  // ãƒ—ãƒ©ãƒƒãƒˆãƒ•ã‚©ãƒ¼ãƒ ãƒ•ã‚£ãƒ«ã‚¿ãƒ¼
  if (platforms.length > 0) {
    dbQuery = dbQuery.overlaps('platforms', platforms)
  }

  // ã‚½ãƒ¼ãƒˆã¨åˆ¶é™
  dbQuery = dbQuery
    .order('metascore', { ascending: false })
    .limit(limit)

  const { data, error } = await dbQuery

  if (error) {
    console.error('Search error:', error)
    throw error
  }

  return data
})
